---
title: isEmpty
---
Checks if `value` is an empty object or collection.

## API

```typescript
function isEmpty(value: unknown): boolean;
```

### Usage

```typescript
import { isEmpty } from '@feedzai/js-utilities';

isEmpty(null)
// true

isEmpty('')
// true

isEmpty({})
// true

isEmpty([])
// true

isEmpty({a: '1'})
// false
```
