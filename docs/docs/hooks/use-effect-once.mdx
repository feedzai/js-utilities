---
title: useEffectOnce
---
Runs an effect only once.

## API

```typescript
function useEffectOnce(effect: EffectCallback): void;
```

### Usage

```tsx
import { useEffectOnce } from '@feedzai/js-utilities/hooks';

useEffectOnce(() => {
  console.log('This runs only one time, when mounting');

  return () => {
    console.log('This runs only one time, when unmounting');
  }
});
```
